        function factorize(n) {
            const factors = {};
            let d = 2;
            while (d * d <= n) {
                while (n % d === 0) {
                    factors[d] = (factors[d] || 0) + 1;
                    n /= d;
                }
                d++;
            }
            if (n > 1) factors[n] = 1;
            return factors;
        }
